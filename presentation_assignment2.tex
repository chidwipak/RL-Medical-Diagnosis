\section{Why Model-Free?}

\begin{frame}{Model-Based vs Model-Free}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Model-Based (Assignment 1):}
        \begin{itemize}
            \item Needs $P(s'|s,a)$ explicitly
            \item Plans using Bellman equations
            \item Fast (no sampling needed)
            \item But requires full model
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{Model-Free (Assignment 2):}
        \begin{itemize}
            \item No $P(s'|s,a)$ needed!
            \item Learns from \textbf{experience} (episodes)
            \item Uses $\epsilon$-greedy exploration
            \item Slower but more general
        \end{itemize}
    \end{columns}

    \vspace{0.5cm}

    \begin{block}{Key Insight}
        Model-free methods learn Q-values directly by \textbf{interacting} with the environment, not by computing them from a known model.
    \end{block}
\end{frame}

\begin{frame}{Why Can't We Use Greedy Policy?}
    \begin{block}{The Exploration Problem}
        In model-free RL, we learn from experience. If we always pick the \textbf{best known} action (greedy), we will never try other actions!
    \end{block}

    \vspace{0.3cm}

    \textbf{Solution: $\epsilon$-Greedy Policy}
    \begin{itemize}
        \item With probability $\epsilon$: pick a \textbf{random} action (explore)
        \item With probability $1-\epsilon$: pick the \textbf{best} action (exploit)
    \end{itemize}

    \vspace{0.3cm}

    \textbf{GLIE Condition}
    \begin{itemize}
        \item \textbf{G}reedy in the \textbf{L}imit with \textbf{I}nfinite \textbf{E}xploration
        \item $\epsilon$ decays over time: $\epsilon_k = \frac{1}{1 + k / N}$
        \item Explores enough early, then converges to optimal
    \end{itemize}
\end{frame}

\section{Monte Carlo Methods}

\begin{frame}{Monte Carlo Prediction — Learning V(s)}
    \begin{block}{Idea}
        Estimate $V(s)$ by averaging the returns observed after visiting state $s$.
    \end{block}

    \vspace{0.3cm}

    \textbf{First-Visit Monte Carlo:}
    \begin{enumerate}
        \item Generate a complete episode: $s_0, a_0, r_1, s_1, a_1, r_2, \dots, s_T$
        \item For the \textbf{first} time state $s$ appears in episode:
        \item Compute return: $G_t = r_{t+1} + \gamma r_{t+2} + \gamma^2 r_{t+3} + \dots$
        \item Update: $V(s) \leftarrow V(s) + \frac{1}{N(s)}(G_t - V(s))$
    \end{enumerate}

    \vspace{0.3cm}

    \textbf{Key Property:} No bootstrapping — uses \textbf{actual} complete returns, not estimates.
\end{frame}

\section{Temporal Difference Methods}

\begin{frame}{One-Step TD (TD(0)) Prediction --- Learning V(s) Step by Step}
    \begin{block}{Idea}
        Update $V(s)$ after \textbf{every single step}, using the estimated value of the next state. This is called \textbf{One-Step TD} or equivalently \textbf{TD(0)} because it looks only 1 step ahead.
    \end{block}

    \vspace{0.3cm}

    \textbf{One-Step TD Update Rule:}
    \[V(s) \leftarrow V(s) + \alpha \underbrace{[r + \gamma V(s') - V(s)]}_{\text{TD Error } \delta}\]

    \vspace{0.3cm}

    \textbf{Key Differences from Monte Carlo:}
    \begin{itemize}
        \item Updates \textbf{every step} (not end of episode)
        \item \textbf{Bootstraps}: uses $V(s')$ as estimate of future return
        \item Lower variance but introduces some bias
        \item Converges faster in practice
    \end{itemize}
\end{frame}

\begin{frame}{TD($\lambda$) --- Forward View: $\lambda$-Return}
    \begin{block}{Key Idea}
        TD($\lambda$) computes a \textbf{weighted average} of all $n$-step returns, from 1-step TD all the way to Monte Carlo.
    \end{block}

    \vspace{0.2cm}

    \textbf{$n$-step Return:} Look $n$ steps ahead:
    \[G_t^{(n)} = r_{t+1} + \gamma r_{t+2} + \dots + \gamma^{n-1} r_{t+n} + \gamma^n V(s_{t+n})\]

    \textbf{$\lambda$-Return:} Weighted average of \textbf{all} $n$-step returns:
    \[G_t^\lambda = (1-\lambda) \sum_{n=1}^{\infty} \lambda^{n-1} G_t^{(n)}\]

    \textbf{The weights $(1-\lambda)\lambda^{n-1}$} form a geometric distribution that sums to 1:
    \begin{itemize}
        \item 1-step gets weight $(1-\lambda)$ \hfill (highest)
        \item 2-step gets weight $(1-\lambda)\lambda$
        \item $n$-step gets weight $(1-\lambda)\lambda^{n-1}$ \hfill (decays geometrically)
    \end{itemize}
\end{frame}

\begin{frame}{TD($\lambda$) --- Backward View: Eligibility Traces}
    \begin{block}{Equivalent Implementation}
        Instead of computing the $\lambda$-return (forward view), we use \textbf{eligibility traces} (backward view) which is computationally efficient and gives the same result.
    \end{block}

    \vspace{0.2cm}

    \textbf{Eligibility Trace:} $E(s)$ tracks how ``recently'' state $s$ was visited.

    \vspace{0.2cm}

    \textbf{Update Rules (Backward View):}
    \begin{enumerate}
        \item Compute TD error: $\delta_t = r_{t+1} + \gamma V(s_{t+1}) - V(s_t)$
        \item Increment trace: $E(s_t) \leftarrow E(s_t) + 1$
        \item Update \textbf{all} states: $V(s) \leftarrow V(s) + \alpha \delta_t E(s)$
        \item Decay traces: $E(s) \leftarrow \gamma \lambda E(s)$
    \end{enumerate}

    \vspace{0.2cm}

    \textbf{$\lambda$ controls the trade-off:}
    \begin{itemize}
        \item $\lambda = 0$: Only 1-step return $\rightarrow$ same as One-Step TD
        \item $\lambda = 1$: Full return $\rightarrow$ same as Monte Carlo
        \item $\lambda = 0.8$: Good balance (what we use)
    \end{itemize}
\end{frame}

\section{Model-Free Control Algorithms}

\begin{frame}{From Prediction to Control}
    \begin{block}{The Control Problem}
        Prediction: Learn $V(s)$ for a given policy.\\
        Control: Find the \textbf{optimal policy} $\pi^*$.
    \end{block}

    \vspace{0.3cm}

    \textbf{Why Q-values instead of V-values?}
    \begin{itemize}
        \item $V(s)$ tells us how good a state is
        \item But to pick the best action, we need: $\pi(s) = \arg\max_a [R + \gamma \sum P(s'|s,a) V(s')]$
        \item This requires $P(s'|s,a)$ — which we don't have!
        \item \textbf{Solution:} Learn $Q(s,a)$ directly → $\pi(s) = \arg\max_a Q(s,a)$
    \end{itemize}

    \vspace{0.3cm}

    \textbf{Three Methods for Action-Value Learning:}
    \begin{enumerate}
        \item MC for Q(s,a) → \textbf{GLIE Monte Carlo Control}
        \item One-Step TD for Q(s,a) \textrightarrow\ \textbf{SARSA}
        \item TD($\lambda$) for Q(s,a) → \textbf{SARSA($\lambda$)}
    \end{enumerate}
\end{frame}

\section{GLIE Monte Carlo Control}

\begin{frame}{GLIE Monte Carlo Control — Algorithm}
    \begin{block}{How It Works}
        Use $\epsilon$-greedy to generate episodes, then update Q(s,a) from complete returns.
    \end{block}

    \vspace{0.2cm}

    \textbf{Algorithm:}
    \begin{enumerate}
        \item Initialize $Q(s,a) = 0$, $N(s,a) = 0$ for all $(s,a)$
        \item For each episode $k$:
        \begin{enumerate}
            \item Set $\epsilon_k = \frac{1}{1 + k/500}$ (GLIE decay)
            \item Generate episode using $\epsilon$-greedy policy
            \item For each first-visit $(s_t, a_t)$ in episode:
                \begin{itemize}
                    \item Compute return: $G_t = \sum_{i=0}^{T-t-1} \gamma^i r_{t+i+1}$
                    \item $N(s_t, a_t) \leftarrow N(s_t, a_t) + 1$
                    \item $Q(s_t, a_t) \leftarrow Q(s_t, a_t) + \frac{1}{N(s_t,a_t)}(G_t - Q(s_t, a_t))$
                \end{itemize}
        \end{enumerate}
        \item Extract policy: $\pi(s) = \arg\max_a Q(s,a)$
    \end{enumerate}
\end{frame}

\section{SARSA}

\begin{frame}{SARSA — Algorithm}
    \begin{block}{How It Works}
        Update Q(s,a) after \textbf{every step} using One-Step TD.
        Name: \textbf{S}tate-\textbf{A}ction-\textbf{R}eward-\textbf{S}tate-\textbf{A}ction.
    \end{block}

    \vspace{0.2cm}

    \textbf{Update Rule:}
    \[Q(s,a) \leftarrow Q(s,a) + \alpha \underbrace{[r + \gamma Q(s', a') - Q(s,a)]}_{\text{TD Error } \delta}\]

    \vspace{0.2cm}

    \textbf{Algorithm:}
    \begin{enumerate}
        \item Initialize $Q(s,a) = 0$ for all $(s,a)$
        \item For each episode:
        \begin{enumerate}
            \item Start at $s_0$, choose $a_0$ using $\epsilon$-greedy
            \item For each step: take action $a$, observe $r, s'$
            \item Choose $a'$ using $\epsilon$-greedy on $s'$
            \item Update: $Q(s,a) \leftarrow Q(s,a) + \alpha[r + \gamma Q(s',a') - Q(s,a)]$
            \item $s \leftarrow s', a \leftarrow a'$
        \end{enumerate}
    \end{enumerate}
\end{frame}

\begin{frame}{SARSA — Results}
    \begin{columns}
        \column{0.5\textwidth}
        \centering
        % IMAGE: results/model_free/sarsa/convergence.png
        \includegraphics[width=0.95\textwidth, height=0.55\textheight, keepaspectratio]{sarsa_convergence}

        \column{0.5\textwidth}
        \textbf{Observations:}
        \begin{itemize}
            \item Updates every step $\rightarrow$ faster feedback
            \item Bootstraps: uses $Q(s',a')$ as estimate
            \item Lower variance than GLIE MC
            \item Some bias from bootstrapping
            \item Converges faster than MC
        \end{itemize}

        \vspace{0.3cm}
        \textbf{SARSA is on-policy:}\\
        It learns Q-values for the policy it is actually following ($\epsilon$-greedy), not a different policy.
    \end{columns}
\end{frame}

\section{SARSA($\lambda$)}

\begin{frame}{SARSA($\lambda$) — Algorithm with Eligibility Traces}
    \begin{block}{How It Works}
        Like SARSA, but propagates TD error to \textbf{all recently visited} state-action pairs using eligibility traces.
    \end{block}

    \vspace{0.2cm}

    \textbf{Update Rules:}
    \begin{align*}
        \delta_t &= r_{t+1} + \gamma Q(s_{t+1}, a_{t+1}) - Q(s_t, a_t) & \text{(TD error)} \\
        E(s_t, a_t) &\leftarrow E(s_t, a_t) + 1 & \text{(accumulating trace)} \\
        Q(s,a) &\leftarrow Q(s,a) + \alpha \delta_t E(s,a) & \text{(update all pairs)} \\
        E(s,a) &\leftarrow \gamma \lambda E(s,a) & \text{(decay traces)}
    \end{align*}

    \vspace{0.2cm}

    \textbf{Key Idea:} When the AI makes a correct diagnosis ($\delta > 0$), credit goes to \textbf{all} the symptom-asking actions that led to it, proportional to how recently they occurred!
\end{frame}

\begin{frame}{Eligibility Traces — Intuition}
    \begin{block}{What Are Eligibility Traces?}
        A memory of which state-action pairs were recently visited.
    \end{block}

    \vspace{0.3cm}

    \textbf{Example in Medical Diagnosis:}
    \begin{enumerate}
        \item AI asks Fever → $E(\text{s0}, \text{Ask Fever}) = 1$
        \item AI asks Cough → $E(\text{s0}, \text{Ask Fever}) = \gamma\lambda \times 1 = 0.72$
        \item AI diagnoses correctly → Big positive $\delta$
        \item \textbf{Both} ``Ask Fever'' and ``Ask Cough'' get credit!
    \end{enumerate}

    \vspace{0.3cm}

    \textbf{$\lambda = 0.8$, $\gamma = 0.9$:} $\gamma\lambda = 0.72$\\
    Recent actions get more credit than older ones.

    \vspace{0.3cm}

    \textbf{This is why SARSA($\lambda$) converges faster:} Credit is assigned to the entire sequence of helpful actions, not just the last one!
\end{frame}

        \includegraphics[width=0.95\textwidth, height=0.55\textheight, keepaspectratio]{sarsa_lambda_convergence}

            \item $\lambda=0$: Same as SARSA
            \item $\lambda=1$: Same as MC
            \item $\lambda=0.8$: Best of both worlds
        \end{itemize}
    \end{columns}
\end{frame}

\section{Discussion \& Summary}

\begin{frame}{Key Insights}
    \begin{block}{What We Learned}
        Model-free methods can find optimal policies \textbf{without} knowing the environment dynamics!
    \end{block}

    \vspace{0.3cm}

    \textbf{Important Observations:}
    \begin{enumerate}
        \item \textbf{Exploration is crucial}: $\epsilon$-greedy ensures we try all actions
        \item \textbf{MC vs TD trade-off}: MC is unbiased but slow; TD is biased but fast
        \item \textbf{Eligibility traces help}: SARSA($\lambda$) assigns credit to the right actions
        \item \textbf{Same result}: All methods converge to same optimal policy as DP
        \item \textbf{Practical}: Model-free methods work even when we can't compute $P(s'|s,a)$
    \end{enumerate}
\end{frame}

\begin{frame}{Assumptions}
    \begin{block}{Assumptions Made}
        Important assumptions for our implementation.
    \end{block}

    \vspace{0.3cm}

    \begin{enumerate}
        \item \textbf{Stationary Environment}: Disease patterns don't change over time
        \item \textbf{Discount Factor}: $\gamma = 0.9$ balances immediate and future rewards
        \item \textbf{Reward Structure}: Same as Assignment 1 ($-0.1$ ask, $+10$ correct, $-5$ wrong)
        \item \textbf{Uniform Sampling}: Each disease appears with equal probability during training
        \item \textbf{GLIE Schedule}: $\epsilon_k = 1/(1 + k/500)$ ensures sufficient exploration
        \item \textbf{Learning Rate}: $\alpha$ decays slowly to ensure convergence
        \item \textbf{$\lambda = 0.8$}: Chosen for SARSA($\lambda$) as a good balance between MC and TD
    \end{enumerate}
\end{frame}
